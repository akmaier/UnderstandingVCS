% ---- Page 2 : Motivation (segment 2) ----
\begin{frame}{Explanation needs ground truth}
  \begin{itemize}\setlength\itemsep{0.5em}
    \item To check an explanation, we must know the true mechanism
    \item Simple models: mechanism known, but explaining it tests nothing
    \item Deep networks: explanation needed, but no inner ground truth
    \item So an explanation can be confident and wrong
    \item We want a system that is complex, fully known, and differentiable
  \end{itemize}
\end{frame}

% ---- Page 6 : Theorem 1 (segment 8) ----
\begin{frame}{Theorem 1: exact forward equivalence}
  \begin{block}{}
    For every reachable state $x$ and every finite sharpness $\alpha$ and
    temperature $T$,
    \[
      \Phi_{\mathrm{S}}(x) \;=\; \Phi_{\mathrm{H}}(x)
    \]
    in identical float32, so the soft and hard trajectories coincide at every
    step. The modes differ only in the gradient
    $\partial\Phi_{\mathrm{S}}/\partial x$, which is defined and nonzero where the
    hard one is zero or undefined.
  \end{block}
  \textbf{Why it matters.} Attribution is computed on the \emph{exact} hard
  trajectory, not an approximation. The forward error is identically zero, at any
  temperature.
\end{frame}

% ---- Page 7 : Theorem 2 (segment 9) ----
\begin{frame}{Theorem 2: temperature-limit bound}
  \begin{block}{}
    The \emph{fully relaxed} step (softmax dispatch, sigmoid branch,
    distance-softmax read, \emph{without} the straight-through correction)
    converges to the hard step as $T\to0$ and $\alpha\to\infty$:
    \[
      \big\|\Phi^{T}_{\mathrm{S}}(x)-\Phi_{\mathrm{H}}(x)\big\|_\infty
      \le C\big[(K{-}1)\,e^{-\Delta_{\min}/T} + e^{-\alpha m_{\min}} + \rho\,e^{-1/T}\big].
    \]
    The deviation falls exponentially as the temperature drops and the sharpness grows.
  \end{block}
  \textbf{Why it matters.} The relaxation is principled even without the
  straight-through trick. The straight-through estimator reaches this hard limit
  \emph{exactly} at finite temperature.
\end{frame}
